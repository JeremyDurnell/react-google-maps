# Pan To

Changes the center of the map to the given `position`.

If the change is less than both the width and height of the map, the transition will be smoothly animated.

**Google Maps Docs:**

- [google.maps.Map](https://developers.google.com/maps/documentation/javascript/reference#Map)

<Playground>
  <TransitionExample
    zoom={8}
    center={{ lat: 36, lng: -122 }}
    initialState={{ position: null }}
    onAnimateClick={({ setState }) => {
      const latOffset = (Math.random() - 0.5) * 2
      const lngOffset = (Math.random() - 0.5) * 2
      setState({ position: { lat: 36 + latOffset, lng: -122 + lngOffset } })
    }}
    render={({ state }) => (
      <>
        {state.position != null && (
          <>
            <PanTo position={state.position} />
            <Marker label="P" position={state.position} />
          </>
        )}

        <Marker label="C" position={{ lat: 36, lng: -122 }} />
      </>
    )}
  />
</Playground>
